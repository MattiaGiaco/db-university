SELECT COUNT(*) AS `n_teachers`, `office_address`
FROM `teachers`
GROUP BY `office_address`;

3)
SELECT AVG(`vote`) AS `avg_votes`, `exam_id`
FROM `exam_student`
GROUP BY `exam_id`;

4)
SELECT COUNT(*) AS `n_degrees`, `department_id`
FROM `degrees`
GROUP BY `department_id`;
